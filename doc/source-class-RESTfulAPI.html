205:   /**
206:    * Handles authentications methods
207:    * get response from API Authenticator
208:    * then passes it on to $answer()
209:    * 
210:    * @param  SS_HTTPRequest $request HTTP request
211:    */
212:   public function auth(SS_HTTPRequest $request)
213:   {
214:     $action = $request->param('Action');
215: 
216:     if ( $this->authenticator )
217:     {
218:       $className = get_class($this->authenticator);
219:       $allowedActions = Config::inst()->get( $className, 'allowed_actions', Config::INHERITED );
220:       if ( !$allowedActions )
221:       {
222:         $allowedActions = array();
223:       }
224: 
225:       if ( in_array($action, $allowedActions) )
226:       {
227:         if ( method_exists($this->authenticator, $action) )
228:         {
229:           $response = $this->authenticator->$action($request);
230:           $response = $this->serializer->serialize( $response );
231:           $this->answer($response);
232:         }
233:         else{
234:           $this->answer(null, array(
235:             'code' => 404,
236:             'description' => "Action '$action' isn't available on class $className."
237:           ));
238:         }
239:       }
240:       else{
241:         $this->answer(null, array(
242:           'code' => 403,
243:           'description' => "Action '$action' not allowed."
244:         ));
245:       }
246:       
247:     }
248:   }

303:   function answer( $json = null, $error = false, $corsPreflight = false )
304:   {
305:     $answer = new SS_HTTPResponse();
306: 
307:     //set response body
308:     if ( !$corsPreflight )

310:       $answer->setBody($json); 
311: 
312:       //Set status code+descript, i.e. 403 Access denied
313:       if ( $error !== false )

315:         $answer->setStatusCode($error['code'], $error['description']);
316:       }     
317:     }
318: 
319:     //set CORS if needed
320:     $answer = $this->setAnswerCORS( $answer );

322:     $answer->addHeader('Content-Type', $this->serializer->getcontentType() );

330:   /**
331:    * Apply the proper CORS response heardes
332:    * to an SS_HTTPResponse
333:    * 
334:    * @param SS_HTTPResponse $answer The updated response if CORS are neabled
335:    */
336:   private function setAnswerCORS(SS_HTTPResponse $answer)
337:   {
338:     $cors = Config::inst()->get( 'RESTfulAPI', 'cors', Config::INHERITED );
339: 
340:     // skip if CORS is not enabled
341:     if ( !$cors['Enabled'] )
342:     {
343:       return $answer;
344:     }
345: 
346:     //check if Origin is allowed
347:     $allowedOrigin = $cors['Allow-Origin'];
348:     $requestOrigin = $this->request->getHeader('Origin');
349:     if ( $requestOrigin )
350:     {
351:       if ( $cors['Allow-Origin'] === '*' )
352:       {
353:         $allowedOrigin = $requestOrigin;
354:       }
355:       else if ( is_array($cors['Allow-Origin']) )
356:       {
357:         if ( in_array($requestOrigin, $cors['Allow-Origin']) )
358:         {
359:           $allowedOrigin = $requestOrigin;
360:         }
361:       }
362:     }
363:     $answer->addHeader('Access-Control-Allow-Origin', $allowedOrigin);
364:     
365:     //allowed headers
366:     $allowedHeaders = '';
367:     $requestHeaders = $this->request->getHeader('Access-Control-Request-Headers');  
368:     if ( $cors['Allow-Headers'] === '*' )
369:     {
370:       $allowedHeaders = $requestHeaders;
371:     }
372:     else{
373:       $allowedHeaders = $cors['Allow-Headers'];
374:     }
375:     $answer->addHeader('Access-Control-Allow-Headers', $allowedHeaders);
376: 
377:     //allowed method
378:     $answer->addHeader('Access-Control-Allow-Methods', $cors['Allow-Methods']);
379: 
380:     //max age
381:     $answer->addHeader('Access-Control-Max-Age', $cors['Max-Age']);
382: 
383:     return $answer;
384:   }
